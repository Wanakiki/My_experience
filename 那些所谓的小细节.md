```cpp
#include <iostream>
using namespace std;
bool isprime(int);
int main()
{
    int Num;
    cout<<"Please enter a Number:";
    cin>>Num;
    for(int i=2;i<=Num;i++)
    {
        if(Num % i==0 && isprime(i))
        {
            cout<<i<<"  ";
        }
    }
    //system("PAUSE");
return 0;
}
bool isprime(int n)
{
    if(n<2)
        return false;
    else
    {
        for(int i=2;i<n-1;i++)
        {
            if(n % i ==0)
                return false;
        }
        return true;
    }
}
```
